85 __attribute__((malloc,warn_unused_result))
86 struct MemNode* alloc_poolable_mem(struct MemPool* pool) {
87  assert(pool != NULL);
88  assert(pool->mem_size != 0);
89 
90  struct MemNode* node = malloc(sizeof(struct MemNode));
91  if (node == NULL) {
92  return NULL;
93  }
94 
95  node->next = NULL;
96  ptrdiff_t* ptr_data = malloc(pool->mem_size + sizeof(ptrdiff_t));
97 
98  if(ptr_data == NULL) {
99  free(node);
100  return NULL;
101  }
102 
103  // store the address of the owning node in the first sizeof(ptrdiff_t) byes
104  ptr_data[0] = (ptrdiff_t) node;
105  node->data = ptr_data+1;
106  node->pool = pool;
107  return node;
108 }

118  assert(node != NULL);

120  // we need to restore all the allocated memory
121  ptrdiff_t* ptr_data = node->data;
122  void* node_data = ptr_data-1;

206 __attribute__((warn_unused_result))
207 void* pool_mem_acquire(struct MemPool* pool) {
208  assert(pool != NULL);
209 
210  while (true) {
211  struct MemNode* previous_head = __atomic_load_n(&pool->head, __ATOMIC_ACQUIRE);
212  if (previous_head == NULL) {
213  return pool_mem_try_alloc_data(pool);
214  }
215 
216  struct MemNode* new_head = __atomic_load_n(&previous_head->next,__ATOMIC_ACQUIRE);
217  if (__atomic_compare_exchange_n(&pool->head, &previous_head, new_head, true, __ATOMIC_SEQ_CST, __ATOMIC_SEQ_CST)) {
218  track_pool_usage_memnode_unavailable(pool);
219  return previous_head->data;
220  }
221  }
222 }

229 void pool_mem_return(void* data) {
230  assert(data != NULL);
231 
232  struct MemNode* new_head = get_memnode_in_data(data);
233  struct MemPool* pool = new_head->pool;
234 
235  while (true) {
236  struct MemNode* previous_head = __atomic_load_n(&pool->head, __ATOMIC_ACQUIRE);
237  __atomic_store_n(&new_head->next, previous_head, __ATOMIC_RELEASE);
238  if (__atomic_compare_exchange_n(&pool->head, &previous_head, new_head, true, __ATOMIC_SEQ_CST, __ATOMIC_SEQ_CST)) {
239  track_pool_usage_returned(pool);
240  return;
241  }
242  }

313 uint32_t find_multipool_index_for_size(uint32_t size) {
314  static const uint32_t int32bitcount = sizeof(uint32_t) * CHAR_BIT;
315  uint32_t size_value = (size - 1) >>DynPoolMinMultiPoolMemNodeSizeBits;
316  if( __builtin_expect(size_value == 0, 0)) {
317  return 0;
318  }
319  return (int32bitcount - (uint32_t)__builtin_clz(size_value));
320 }
