65  static auto WrapOnOverflow(std::atomic<IndexSizeT>& position, IndexSizeT size) -> void {
66  auto curPos = position.load(std::memory_order_acquire);
67 
68  while(curPos==size) {
69  if(position.compare_exchange_weak(curPos, 0, std::memory_order_acq_rel)) {
70  return;
71  }
72  curPos = position.load(std::memory_order_acquire);
73  }

83  ConcurrentIndexer(IndexSizeT poolSize):size(poolSize*2), maxPositionSize((std::numeric_limits<IndexSizeT>::max()/this->size) * this->size) {
84  // The idea is to have all possible indices given a max pool size tracked in this vector
85  // when an element is requested from the pool, the index of the item in the pool is retrieved by returning
86  // an index from indices starting from the read position.
87  // However, the vector size never changes, only the position of the next index is incremented or decremented.
88  // The order of elements in the vector is not important so long as it has unique elements.
89 
90  // As indices are being consumed, the read position moves but marker for when we can't read anymore only advances
91  // when there are no more indices to read.
92  // The same thing happens to returned indices, the write position moves but the end position only advances when we can't write anymore.
93  // One important design decision, is that we don't check if we can write over valid read positions since that should never happen
94  // if only consumed indices are returned. That's intentional for simplicity and performance reasons.
95 
96  this->indices.reserve(this->size);
97  this->indices.assign(this->size, UnusedPosition);
98  // populate the entire vector with values but only the first half are used for indices
99  // this is just so we properly allocate all positions
100  for(IndexSizeT i = 0; i < poolSize; ++i) {
101  // add one to the indice so we can reserve 0 as UnusedPosition
102  this->indices[i] = i+1;
103  }
104 
105 
106  // set the write position one indice after the end of all valid indices
107  writePos = poolSize;
108  }

116  auto Next() -> IndexHolder {
117  while(true) {
118  // do we always need to load? perhaps not but, better be safe when it comes to atomic operations. I'll save the regrets for later
119  IndexSizeT curReadPos = this->readPos.load(std::memory_order_acquire);
120 
121  if(curReadPos == this->maxPositionSize) {
122  WrapOnOverflow(this->readPos, this->maxPositionSize);
123  } else {
124  IndexSizeT curWritePos = this->writePos.load(std::memory_order_acquire);
125 
126  // There's no more items to read because we read all nothing has been returned
127  if(curReadPos == curWritePos) {
128  return {};
129  }
130 
131  // The action read index needs to be constrained to the size of the buffer
132  // Nothing prevents two threads from having the same read index for differents read positions
133  // because other threads could keep moving the read position forward
134  // so we try to correct that later when reading the value from the vector
135  const IndexSizeT curReadIndex = curReadPos % this->size;
136 
137  // It's possible that writing (Return) has started (incremented writePos) but the value hasn't been written yet
138  // Note to self: this should never happen and if it does there might be a bug somewhere
139  if(unlikely(AtomicLoad(&this->indices[curReadIndex]) == UnusedPosition)) {
140  return {};
141  }
142 
143  auto modified = this->readPos.compare_exchange_weak(curReadPos, curReadPos + 1, std::memory_order_acq_rel);
144 
145  if(modified) {
146  IndexSizeT index = AtomicLoad(&this->indices[curReadIndex]);
147  // It's possible we got here because we wrapped but the thread writing to this position
148  // hasn't finished doing so yet. We try to keep the behaviour correct via busy wait
149  // This situation could happen if the size is much smaller than the number of threads calling it
150  while(index == UnusedPosition) {
151  index = AtomicLoad(&this->indices[curReadIndex]);
152  std::this_thread::yield();
153  }
154 
155  AtomicStore(&this->indices[curReadIndex], UnusedPosition);
156 
157  return {index-1};
158  }
159  }
160  }
161  }

170  auto Return(IndexSizeT index) -> void {
171  while(true) {
172  auto curWritePos = this->writePos.load(std::memory_order_acquire);
173  if(curWritePos == this->maxPositionSize) {
174  WrapOnOverflow(this->writePos, this->maxPositionSize);
175  } else {
176 
177  // Note we don't want calls to this method to ever fail (return without writing) so the write position must always be valid
178  auto modified = this->writePos.compare_exchange_weak(curWritePos, curWritePos+1, std::memory_order_acq_rel);
179 
180  if(modified) {
181  // As in the read position, various threads could get here and curWriteIndex will wrap to the same position in the vector
182  // and we try to correct it later when writing the value back
183  const IndexSizeT curWriteIndex = curWritePos % this->size;
184 
185  while(AtomicLoad(&this->indices[curWriteIndex]) != UnusedPosition) {
186  std::this_thread::yield();
187  }
188 
189  // add back 1 that was subtracted before
190  AtomicStore(&this->indices[curWriteIndex], index+1);
191 
192 
193  return;
194  }
195  }
196  }
